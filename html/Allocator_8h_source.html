83  // while p is not the last sentinel, check if the sentinels are valid
84  while (p < &(*this)[N - sizeof(int)]) {

139  pointer allocate(size_type num_obj) {
140  if (num_obj <= 0) {
141  std::bad_alloc exception;
142  throw exception;
143  }
144 
145  // number of bytes to allocate
146  size_type size = num_obj * sizeof(T);
147 
148  // initialize p to the first sentinel
149  int *p = &(*this)[0];
150 
151  // advance p until we find the right block or we hit the end
152  while (p != &(*this)[N - sizeof(int)]) {
153  if (*p > 0 && static_cast<unsigned int>(*p) >= size) {
154  break;
155  }
156 
157  // add 2 to account for the sentinels
158  p += std::abs(*p) / sizeof(int) + 2;
159  }
160 
161  // if we exit the loop, and the block is too small (or negative)
162  // return NULL
163  if (*p < 0 || (*p > 0 && static_cast<unsigned int>(*p) < size))
164  return 0;
165 
166  // pointer to beginning of user's block
167  int *new_block = p + 1;
168 
169  int free_size = *p;
170 
171  // if there isn't enough space left over for another,
172  // take the whole block
173  if (static_cast<unsigned int>(free_size - size) < min_size) {
174  *p *= -1;
175  *(p + free_size / sizeof(int) + 1) *= -1;
176  val_pointers.insert(new_block);
177  return reinterpret_cast<pointer>(new_block);
178  }
179 
180  int *start_point = p;
181  int *end_point = start_point + size / sizeof(int) + 1;
182  *p = -size;
183  *end_point = -size;
184 
185  // only update the following sentinels if we aren't at the end
186  if (end_point != &(*this)[N - sizeof(int)]) {
187  p = end_point + 1;
188  end_point = start_point + free_size / sizeof(int) + 1;
189 
190  int new_free_size = free_size - size - (2 * sizeof(int));
191  *p = new_free_size;
192  *end_point = new_free_size;
193  }
194 
195  val_pointers.insert(new_block);
196  return reinterpret_cast<pointer>(new_block);
197  }

224  void deallocate(pointer p, size_type) {
225  int *point = reinterpret_cast<int *>(p);
226 
227  if (val_pointers.find(point) == val_pointers.end()) {
228  throw std::invalid_argument("invalid arg in deallocate");
229  } else
230  val_pointers.erase(point);
231 
232  int *front = point - 1;
233  int *end = front + -1 * (*front) / sizeof(int) + 1;
234  int size = -1 * (*front);
235 
236  // update our sentinels according to whether or not we
237  // should coalesce with other free blocks
238  if (front != &(*this)[0] && *(front - 1) > 0) {
239  front -= *(front - 1) / sizeof(int) + 2;
240  size += *front + 2 * sizeof(int);
241  }
242 
243  if (end != &(*this)[N - sizeof(int)] && *(end + 1) > 0) {
244  end += *(end + 1) / sizeof(int) + 2;
245  size += *end + 2 * sizeof(int);
246  }

250  }
